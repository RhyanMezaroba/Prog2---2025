<?php
class Produto {
    public $preco;
    public $desconto;
    
    public function precoFinal() {
        return $this->preco - ($this->preco * $this->desconto);
    }
}

$frase = 'O valor do Refri é ';
$refrigerante = new Produto();
$refrigerante->preco = 100;
$refrigerante->desconto = 0.1;

echo $frase . $refrigerante->precoFinal();
?>
